## Example One: Ticket Management Plugin (Go)

This plugin displays a ticket list in the visitor panel and provides a "Create Ticket" form in the chat toolbar.

### Core Code (`main.go`)

```go
package main

import (
	"fmt"
	"github.com/ivansslo/roadfx-plugin-go"
)

type TicketPlugin struct { roadfx.BasePlugin }

func (p *TicketPlugin) Capabilities() []roadfx.Capability {
	return []roadfx.Capability{
		roadfx.VisitorPanel("Related Tickets", roadfx.WithIcon("ticket")),
		roadfx.ChatToolbar("Create Ticket", roadfx.WithIcon("plus-circle")),
	}
}

func (p *TicketPlugin) OnVisitorPanelRender(ctx *roadfx.RenderContext) roadfx.Template {
	return roadfx.NewGroup().
		Add(roadfx.NewTable("Ticket List").
			Columns("ID", "Title", "Status").
			Row(map[string]any{"ID": "TK-001", "Title": "Cannot Login", "Status": "Open"}))
}

func (p *TicketPlugin) OnChatToolbarEvent(ctx *roadfx.EventContext) *roadfx.Action {
	if ctx.ActionID == "submit" {
		title := ctx.FormData["title"].(string)
		return roadfx.ShowToast(fmt.Sprintf("Ticket '%s' submitted successfully", title), "success").
			Then(roadfx.CloseModal())
	}

	form := roadfx.NewForm("Create Ticket").
		Add(roadfx.NewFormField("title", "Title", "text").SetRequired(true))
	return roadfx.ShowModal("Create Ticket", form)
}

func main() { roadfx.Run(&TicketPlugin{}) }
```

## Example Two: CRM Customer Profile (Python)

This plugin displays customer profiles in the visitor panel by integrating with an enterprise CRM system.

### Core Code (`main.py`)

```python
from roadfx_plugin import ROADFXPlugin, Capability, RenderContext, KeyValue, Group, Action

class CRMPlugin(ROADFXPlugin):
    id = "com.roadfx.crm"
    name = "CRM Customer Profile"

    capabilities = [
        Capability.visitor_panel(title="Customer Profile", icon="user")
    ]

    def on_visitor_panel_render(self, ctx: RenderContext):
        # 模拟从 API 获取数据
        customer_data = {"level": "Gold Member", "total": "¥12,000"}

        ui = Group()
        info = KeyValue(title="Basic Information")
        info.add("Customer Level", customer_data["level"], icon="crown", color="#FFD700")
        info.add("Total Spent", customer_data["total"])
        ui.add(info)
        return ui

if __name__ == "__main__":
    CRMPlugin().run()
```
